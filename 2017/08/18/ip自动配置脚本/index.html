ip自动配置脚本

2017-08-18

无聊写个ip自动配置脚本，能力有限勿喷。可以在此基础上进行完善

#!/bin/bash
# by garywu - 2017.8.18


read -p "请输入IP地址: " ip

net_path=/etc/sysconfig/network-scripts/ifcfg-eth0

if [[ $ip -gt 0 && $ip -le 254 ]];then
   sed -i 's#BOOTPROTO=dhcp#BOOTPROTO=static#g' $net_path
   echo "IPADDR=192.168.10.$ip" >> $net_path
   echo "NETMASK=255.255.255.0" >> $net_path
   echo "GATEWAY=192.168.10.1" >> $net_path
   echo "DNS1=114.114.114.114" >> $net_path
