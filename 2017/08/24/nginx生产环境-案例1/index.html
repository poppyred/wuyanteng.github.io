nginx生产环境-案例1

2017-08-24

研发项目需求：https、环境：php7

80配置

server {
    listen  1.1.1.1:80;
    server_name  xxx.test.net;    #配置主机头；不同主机头可以配置相同端口
    proxy_buffers 4 1m;
    proxy_busy_buffers_size 2m;

    root /var/html/www/xxx.test.net/;       #代码存放目录
    keepalive_timeout 0;

# 屏蔽svn信息
    location ~ ^(.*)\/\.svn\/ {
        deny all;
    }

    location = /favicon.ico {
        access_log off;
        log_not_found off;
    }   

    location ~ \.php$ {

        fastcgi_pass  php5.6;        #定义php版本
        include fastcgi.conf;
    }
    access_log      /var/log/nginx/xxx.test.net.access_log main;
    error_log       /var/log/nginx/xxx.test.net.net.error_log warn;
}

443配置

server {
    listen      1.1.1.1:443;
    server_name xxx.test1.net;        #配置主机头

    root /var/html/www/xxx.test1.net/;   #代码存放目录

    ssl on;
    ssl_certificate         /var/html/ssl/xxx.net.pem;
    ssl_certificate_key     /var/html/ssl/xxx.net.key;
    ssl_session_timeout  5m;
    ssl_protocols  TLSv1 TLSv1.1 TLSv1.2;
    ssl_ciphers  HIGH:!RC4:!MD5:!aNULL:!eNULL:!NULL:!DH:!EDH:!EXP:+MEDIUM;
    ssl_prefer_server_ciphers   on;

# 屏蔽svn信息
    location ~ ^(.*)\/\.svn\/ {
        deny all;
    }

    location = /favicon.ico {
        access_log off;
        log_not_found off;
    }   

    location ~ \.php$ {
        fastcgi_pass  php7.1;      #定义php版本
        include fastcgi.conf;
    }
    access_log      /var/log/nginx/xxx.test1.net.access_log main;
    error_log       /var/log/nginx/xxx.test1.net.net.error_log warn;
}

nginx定义php版本-目的：让配置文件直接调用

vim /etc/nginx/conf.d/php_version.conf

upstream php5.6 
{
 server   unix:/var/run/php5.6-php-fpm1.sock weight=100 max_fails=10 fa    il_timeout=30;
}                   

upstream php7.1 
{   
server   unix:/var/run/php7-php-fpm7.sock  weight=100 max_fails=10     fail_timeout=30;
}

配置文件语法检查

[root@localhost]# nginx -t

nginx: the configuration file /etc/nginx/nginx.conf syntax is ok
nginx: configuration file /etc/nginx/nginx.conf test is successful

优雅重启nginx

/etc/init.d/nginx -s reload

查看日志

tailf  /var/log/nginx/xxx.test1.net.access_log
