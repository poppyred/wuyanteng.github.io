mysql备份脚本-rsync上传

2017-06-01

脚本如下：

#!/bin/sh
# Comman: update OwnCloud Server
# Database info

DB_USER="root"
DB_PASS="passwd"
DATE=$(date +%Y%m%d)
BIN_DIR="/usr/local/mysql/bin"
BCK_DIR="/opt/"

${BIN_DIR}/mysqldump --opt -u${DB_USER} -p${DB_PASS} stfoa > ${BCK_DIR}/oa_mysql_${DATE}.sql
cd ${BCK_DIR}

/usr/bin/rsync -avzP oa_mysql_${DATE}.sql eegs@192.168.10.12::eegs
