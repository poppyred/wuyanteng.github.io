saltstack安装keepalived

2017-10-25

salt安装配置启动keepalived

根据keepalived.sls来配置 salt://cluster/files/haproxy-outside-keepalived.conf 目录下文件选项的调用配置

keepalived.sls

include:
  - keepalived.install
keepalived-server:
  file.managed:
    - name: /etc/keepalived/keepalived.conf
    - source: salt://cluster/files/haproxy-outside-keepalived.conf
    - mode: 644
    - user: root
    - group: root
    - template: jinja       #开启jinja
    {% if grains['fqdn'] == 'saltstack-node1.example.com' %}   #主机名判断
    - ROUTEID: haproxy_ha
    - STATEID: MASTER
    - PRIORITYID: 150
    {% elif grains['fqdn'] == 'saltstack-node2.example.com' %}   #主机名判断
    - ROUTEID: haproxy_ha
    - STATEID: BACKUP
    - PRIORITYID: 100
    {% endif %}
  service.running:
    - name: keepalived
    - enable: True
    - watch:
      - file: keepalived-server   #当source文件改变时，重启服务
