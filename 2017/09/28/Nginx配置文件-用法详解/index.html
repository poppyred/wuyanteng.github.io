Nginx配置文件-详解

全局选项

user  www www;           #定义Nginx运行的用户和用户组

worker_processes  8;   #定义nginx worker进程数。
一般配置为与可用的CPU内核数相等或者配置成"auto"来自动检测。最多开启8个，8个以上性能就不会再提升了，而且稳定性会变的更低。

worker_cpu_affinity 00000001 00000010 00000100 00001000 00010000 00100000 01000000 10000000;
#配置利用8核CPU(默认没有开启)

worker_rlimit_nofile  51200;   #更改worker进程的最大打开文件数限制。
如果没设置的话，这个值为操作系统的限制。设置后你的操作系统和Nginx可以处理比“ulimit -a”的值更多的文件，所以把这个值设高，这样nginx就不会有“too many open files”问题了。

error_log /var/log/nginx/error.log  warn;
全局错误日志及错误日志定义等级，[ debug |info | notice | warn | error | crit ]

pid  /var/run/nginx.pid        #nginx进程文件

events模块

events{  
    use epoll;      #复用客户端线程的轮询方法使用epoll的I/O 模型
    worker_connections 51200;   #每个worker进程打开的最大连接数,与worker_rlimit_nofile一致即可。

HTTP模块

http {
    include     /etc/nginx/mime.types;   #文件扩展名与文件类型映射表,默认保存在conf/mime.types中
    default_type    application/octet-stream;    #默认文件类型
    charset  utf-8;   #设置默认编码

    log_format main   #定义日志格式
        '$remote_addr - $remote_user [$time_local] '
        '"$request" $status $bytes_sent '
        '"$http_referer" "$http_user_agent" '
        '"$gzip_ratio" "$request_time" '
        '"$upstream_addr" "$upstream_status" "$upstream_response_time"';

    server_names_hash_bucket_size 128;   #服务器名字的hash表大小
    client_header_buffer_size 2k;        
    #为客户端请求头分配一个缓冲区,如果大小大于此值，则使用large_client_header_buffers分配的值

    large_client_header_buffers 4 8k;

    client_max_body_size 25M;     
    #设置NGINX能处理的最大请求主体大小。 如果请求大于指定的大小，则NGINX发回HTTP 413（Request Entity too large）错误。 如果服务器处理大文件上传，则该指令非常重要。

    client_body_buffer_size 1024k;  #一般设置1024即可

    fastcgi_connect_timeout 1800;  #指定连接到后端FastCGI的超时时间。
    fastcgi_send_timeout 1800;     #指定向FastCGI传送请求的超时时间（已经完成两次握手后的时间）
    fastcgi_read_timeout 1800;     #nginx中fastcgi读取超时时间
    fastcgi_buffer_size 1024k;     #用于指定读取FastCGI应答第一部分需要用多大的缓冲区
    fastcgi_buffers 32 1024k;      #指定FastCGI的缓冲区大小
    fastcgi_busy_buffers_size 2048k; #默认值是fastcgi_buffers的两倍
    fastcgi_temp_file_write_size 2048k; #写入缓存文件时使用多大的数据块，默认为fastcgi_buffers的两倍

    sendfile on;      #开启高效文件传输模式
    #autoindex on;    #开启目录列表访问，合适下载服务器，默认关闭。
    tcp_nopush on;    #防止网络阻塞
    tcp_nodelay on;   #防止网络阻塞
    keepalive_timeout 60;   #长连接超时时间，单位是秒

    gzip on;                #开启gzip压缩输出
    gzip_min_length  1100;  #最小压缩文件大小
    gzip_buffers     4 32k; #压缩缓冲区
    gzip_comp_level  9;     #压缩等级
    gzip_types   text/plain application/x-javascript text/xml text/css application/json;
    #压缩类型
    gzip_disable "MSIE [1-6]\.(?!.*SV1)";  #不压缩IE

    ignore_invalid_headers    on; #忽略不合法的http头部
    server_tokens off;       #是否在错误页面中显示nginx的版本号

    server {
        listen       127.0.0.1:80;
        server_name  localhost;
        location /NginxStatus {
            stub_status on;
        }
    }

    index index.html, index.php;        #定义首页文件
    include /etc/nginx/conf.d/*.conf;   #引用conf.d目录中的配置以规范与简化管理

}

http中的server模块

cd  /etc/nginx/conf.d/    
单独配置虚拟主机,为简化管理，每个虚拟主机为独立*.conf配置文件

【前端机】vim game.garywu.com.conf

upstream game_garywu {
        server  192.168.50.140 max_fails=3 fail_timeout=10s;
        server  192.168.50.141 max_fails=3 fail_timeout=10s;
        server  192.168.50.142 max_fails=3 fail_timeout=10s;
        }

server {
        listen    1.1.1.1:80;
        server_name game.garywu.com;

        root /usr/share/html/game.garywu.com;    #定义虚拟目录(前端机可以不存在此目录)


        location / {                                                         
          proxy_pass http://game_garywu;   #启用负载均衡，用上面定义的“game_garywu”                         proxy_next_upstream error timeout invalid_header http_500 http_502 http_503 http_504;           #当其中一台返回错误码error/500等错误时，可以分配到下一台服务器程序继续处理，提高访问成功率

          set $proxy_host "game.garywu";   
          # 定义$proxy_host的值为后端定义的servername "game.garywu"  
          proxy_set_header   Host  $proxy_host;  
          #替换用户请求数据包头部的Host信息为$proxy_host值信息                 
          proxy_set_header   X-Forwarded-For  $remote_addr;   
          #设置显示访问客户端的真实源IP地址         
        }

        location = /favicon.ico {        #定义favicon.ico文件
                access_log off;          #不记录ico访问日志
                log_not_found off;       #不记录ico访问日志

        }



        access_log      /var/log/nginx/game.garywu.com.access_log main;  #定义访问日志
        error_log       /var/log/nginx/game.garywu.com.error_log warn;   #定义错误日志
}

【后端机】vim game.garywu.com.conf

以第一台后端机为例，其他后端机需修改listen IP

server {
    listen          192.168.50.140:80;     #后端机IP地址
    server_name     game.garywu;           #与前端机定义的头部host信息保持一致

    root /usr/share/html/game.garywu.com;  #定义虚拟目录(必须存在且有源码等)

    location ~ ^(.*)\/\.svn\/ {            #漏洞预防-过滤.*和.svn
        deny all;
    }

    location = /favicon.ico {              #定义ico
        access_log off;
        log_not_found off;
    }   

    location ~ \.php$ {    
        include fastcgi_params;
        fastcgi_param SCRIPT_FILENAME /PHP文件路径$fastcgi_script_name;   #PHP文件路径
        #fastcgi_pass  127.0.0.1:9000;  #PHP-FPM地址和端口号
        fastcgi_pass  unix:/var/run/php5-fpm.sock;   #可以使用php-fpm的socket
        #fastcgi_pass  php-garywu;       #也可以指定upstream字段名称（如下） 
    }
        access_log      /var/log/nginx/game.garywu.com.access_log main;  #定义访问日志
        error_log       /var/log/nginx/game.garywu.com.error_log warn;   #定义错误日志
}

配置中将.php结尾的请求通过FashCGI交给PHP-FPM处理，PHP-FPM是PHP的一个FastCGI管理器。

[PHP-FPM]在conf.d目录下创建文件：vim php-garywu.conf

upstream php-garywu  {
  server  unix:/var/run/php5-fpm.sock weight=100 max_fails=10 fail_timeout=30;
}

upstream php7-garywu  {
server   unix:/var/run/php7-fpm.sock weight=100 max_fails=10 fail_timeout=30;
}
