配置触发消息脚本
# vi /etc/wechat.sh

#微信企业号的CropID
CropID='xxxxxxxxxxxx'
#企业号应用Secret
Secret='xxxxxxxxxxxxxxxxxxxxxx'
GURL="https://qyapi.weixin.qq.com/cgi-bin/gettoken?corpid=$CropID&corpsecret=$Secret"
Gtoken=$(/usr/bin/curl -s -G $GURL |  awk -F "[\":,]" '{print $15}')
PURL="https://qyapi.weixin.qq.com/cgi-bin/message/send?access_token=$Gtoken"
function body() {
#Appid 填写企业号中建立的报警APP的ID
local int AppID=1
#此处填写报警接收用户，全部报警可留空
local UserID="@all"
local PartyID="@all"
local TagID="@all"
Ent=$'\n'
#应cactifans群内要求，添加Cacti微信报警日期参数
Date=$(date '+%Y年%m月%d日 %H:%M:%S\n\n')
#读取/tmp/wechat.txt文件中内容到变量Tit
Tit=$(cat /tmp/wechat.txt)
#拼接msg主体文件,包含日期,主题,报警内容.并删除报警内容中的'%'号.
Msg=$Date$Tit$Ent$(cat /tmp/wechat_message.txt|sed 's/%//g')
#获取wechat_message.txt中的url行内容
Url=$(grep  "http" /tmp/wechat_message.txt|sed  's/URL: //g')
Pic_tmp=$(grep  "http" /tmp/wechat_message.txt|sed  's/URL: //g'|sed 's/\/graph.php/\/graph_image.php/g')

#在此修改图片大小，防止图像显示不全,并判断图片文件是否存在。防止宕机出现图片报警
if [ ! -n "$Pic_tmp" ] ;then
    Pic=""
else
    Pic=$Pic_tmp$'&graph_height=215&graph_width=424'
fi
printf '{\n'
printf '\t"touser": "'"$UserID"\"",\n"
printf '\t"toparty": "'"$PartyID"\"",\n"
printf '\t"totag": "'"$TagID"\"",\n"
printf '\t"msgtype": "news",\n'
printf '\t"agentid": "'" $AppID "\"",\n"
printf '\t"news": {\n'
printf '\t"articles": [\n'
printf '{\n'
printf '\t\t"title": "'"$Tit"\","\n"
printf '\t\t"description": "'"$Msg"\","\n"
printf '\t\t"url": "'"$Url"\","\n"
printf '\t\t"picurl": "'"$Pic"\","\n"
printf '\t}\n'
printf '\t]\n'
printf '\t}\n'
printf '}\n'
}
curl -l -H "Content-type: application/json" -X POST -d "$(body )" $PURL
